record MyTheme : LumexTheme
{
    public MyTheme()
    {
        var layoutConfig = new LayoutConfig()
        {
            DisabledOpacity = 0.4,
            Radius = new BaseScale()
            {
                Sm = "0.125rem",
                Md = "0.25rem",
                Lg = "0.375rem"
            }
        };

        Light = new ThemeConfigLight
        {
            Layout = layoutConfig
        };

        Dark = new ThemeConfigDark
        {
            Layout = layoutConfig
        };
    }
}
