<LumexThemeProvider />

@*
    Apply theme tokens on components using different parameters (e.g., `Color`, `Radius`, and etc).
    ThemeColor.Primary => "bg-primary text-primary-foreground"
*@
<LumexButton Color="@ThemeColor.Primary">Button</LumexButton>

@* Apply theme tokens directly on a native HTML element *@
<button class="bg-primary text-primary-foreground">Button</button>

@* 
    Apply theme tokens directly on a LumexUI component.
    ThemeColor.Primary will be effectively overridden by the `Class` parameter
*@
<LumexButton Color="@ThemeColor.Primary"
             Class="bg-danger text-danger-foreground">
    Button
</LumexButton>

@* Apply any Tailwind CSS tokens on both *@
<LumexButton Class="bg-indigo-600 text-zinc-50">Button</LumexButton>
<button class="bg-indigo-600 text-zinc-50">Button</button>
